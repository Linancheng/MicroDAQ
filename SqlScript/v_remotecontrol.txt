CREATE VIEW v_remoteControl AS

SELECT
id,
Min(cycle) AS cycle,
Min(command) AS command,
Min(cmdstate) AS cmdState

FROM
(
SELECT     p.alarm AS id,  (0 | null) AS cycle, 1 AS command, 1 AS cmdState
FROM         dbo.meters_value m INNER JOIN
                      dbo.ProcessItem p ON m.id = p.slave
) b
GROUP BY B.id
HAVING B.ID IS NOT NULL
